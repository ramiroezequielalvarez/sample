  myFilter10(date: Date | null): boolean {
    date = date ?? new Date();
    const dayOfMonth = date.getTime();

    // If tengo date. Le asigno y hago el return completo
    const lastDayOfMonth = new Date('09/10/2023').getTime();

    let d = new Date();
    d.setDate(d.getDate() - 1);
    // Si no tengo date. Solo hago el >
    //return dayOfMonth === lastDayOfMonth || date > new Date();
    return dayOfMonth === lastDayOfMonth || date > d;
    //
  }
